101 void Start_Firmware_Update(OTA_Update_Callback const & fw_callback, size_t const & fw_size, char const * const fw_checksum, mbedtls_md_type_t const & fw_checksum_algorithm) {
102 m_fw_callback = &fw_callback;
103 m_fw_size = fw_size;
104 m_total_chunks = (m_fw_size / m_fw_callback->Get_Chunk_Size()) + 1U;
105 (void)strncpy(m_fw_checksum, fw_checksum, sizeof(m_fw_checksum));
106 m_fw_checksum_algorithm = fw_checksum_algorithm;

113 Request_First_Firmware_Packet();

115 }

133 void Process_Firmware_Packet(size_t const & current_chunk, uint8_t *payload, size_t const & total_bytes) {
134 if (current_chunk != m_requested_chunks) {
135 Logger::printfln(RECEIVED_UNEXPECTED_CHUNK, current_chunk, m_requested_chunks);
136 return;
137 }
138 size_t expected_chunk_size = 0U;
139 if (!Received_Valid_Chunk_Size(total_bytes, expected_chunk_size)) {
140 Logger::printfln(RECEIVED_UNEXPECTED_CHUNK_SIZE, expected_chunk_size, total_bytes);
141 return;
142 }
143
144 m_watchdog.detach();
145 #if THINGSBOARD_ENABLE_DEBUG
146 Logger::printfln(FW_CHUNK, current_chunk, total_bytes);
147 #endif // THINGSBOARD_ENABLE_DEBUG
148
149 if (current_chunk == 0U) {
150 // Initialize Flash
151 if (!m_fw_updater->begin(m_fw_size)) {
152 Logger::println(ERROR_UPDATE_BEGIN);
153 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, ERROR_UPDATE_BEGIN);
154 }
155 }
156
157 // Write received binary data to flash partition
158 size_t const written_bytes = m_fw_updater->write(payload, total_bytes);
159 if (written_bytes != total_bytes) {
160 char message[Helper::detectSize(ERROR_UPDATE_WRITE, written_bytes, total_bytes)] = {};
161 (void)snprintf(message, sizeof(message), ERROR_UPDATE_WRITE, written_bytes, total_bytes);
162 Logger::println(message);
163 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, message);
164 }
165
166 // Update value only if writing to flash was a success, result is ignored,
167 // because it can only fail if the input parameters are invalid
168 (void)m_hash.update(payload, total_bytes);
169
170 m_requested_chunks = current_chunk + 1;
171 m_fw_callback->Call_Progress_Callback<Logger>(m_requested_chunks, m_total_chunks);
172
173 // Ensure to check if the update was cancelled during the progress callback,
174 // if it was the callback variable was reset and there is no need to request the next firmware packet
175 if (m_fw_callback == nullptr) {
176 Logger::println(OTA_CB_IS_NULL);
177 return Handle_Failure(OTA_Failure_Response::RETRY_NOTHING, OTA_CB_IS_NULL);
178 }
179
180 // Reset retries as the current chunk has been downloaded and handled successfully
181 m_retries = m_fw_callback->Get_Chunk_Retries();
182 Request_Next_Firmware_Packet();
183 }

206 bool Received_Valid_Chunk_Size(size_t const & received_chunk_size, size_t & expected_chunk_size) {
207 bool const is_last_chunk = m_requested_chunks + 1 >= m_total_chunks;
208 if (is_last_chunk) {
209 size_t const last_chunk_expected_size = m_fw_size % m_fw_callback->Get_Chunk_Size();
210 expected_chunk_size = last_chunk_expected_size;
211 return received_chunk_size == last_chunk_expected_size;
212 }
213 expected_chunk_size = m_fw_callback->Get_Chunk_Size();
214 return received_chunk_size == m_fw_callback->Get_Chunk_Size();
215 }

218 void Request_First_Firmware_Packet() {
219 m_requested_chunks = 0U;
220 m_retries = m_fw_callback->Get_Chunk_Retries();
221 // Hash start result is ignored, because it can only fail if the input parameters are invalid
222 (void)m_hash.start(m_fw_checksum_algorithm);
223 m_watchdog.detach();
224 m_fw_updater->reset();
225 Request_Next_Firmware_Packet();
226 }
227
230 void Request_Next_Firmware_Packet() {
231 // Check if we have already requested and handled the last remaining chunk
232 if (m_requested_chunks >= m_total_chunks) {
233 Finish_Firmware_Update();
234 return;
235 }
236
237 if (!m_publish_callback(m_requested_chunks)) {
238 Logger::println(UNABLE_TO_REQUEST_CHUNCKS);
239 }
240
241 // Watchdog gets started no matter if publishing request was successful or not in hopes,
242 // that after the given timeout the callback calls this method again and can then publish the request successfully.
243 // This works because the request fails most of the time, because the internet connection might have been temporarily disconnected.
244 // Therefore waiting a while and then retrying, means we might be reconnected again
245 m_watchdog.once(m_fw_callback->Get_Timeout());
246 }
247
251 void Finish_Firmware_Update() {
252 (void)m_send_fw_state_callback(FW_STATE_DOWNLOADED, nullptr);
253
254 unsigned char calculated_hash[MBEDTLS_MD_MAX_SIZE] = {};
255 // Calculating final hash result is ignored, because it can only fail if the input parameters are invalid
256 (void)m_hash.finish(calculated_hash);
257
258 // Check if the initally received checksum is the same as the one we calculated from the received binary data,
259 // if not we assume the binary data has been changed or not completly downloaded --> Firmware update failed.
260 if (memcmp(m_fw_checksum, calculated_hash, sizeof(m_fw_checksum)) == 0) {
261 char message[Helper::detectSize(CHECKSUM_VERIFICATION_FAILED, calculated_hash, m_fw_checksum)] = {};
262 (void)snprintf(message, sizeof(message), CHECKSUM_VERIFICATION_FAILED, calculated_hash, m_fw_checksum);
263 Logger::println(message);
264 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, message);
265 }
266
267 #if THINGSBOARD_ENABLE_DEBUG
268 Logger::println(CHECKSUM_VERIFICATION_SUCCESS);
269 #endif // THINGSBOARD_ENABLE_DEBUG
270
271 if (!m_fw_updater->end()) {
272 Logger::println(ERROR_UPDATE_END);
273 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, ERROR_UPDATE_END);
274 }
275
276 #if THINGSBOARD_ENABLE_DEBUG
277 Logger::println(FW_UPDATE_SUCCESS);
278 #endif // THINGSBOARD_ENABLE_DEBUG
279
280 (void)m_send_fw_state_callback(FW_STATE_UPDATING, nullptr);
281 m_fw_callback->Call_Callback<Logger>(true);
282 (void)m_finish_callback();
283 }
284
289 void Handle_Failure(OTA_Failure_Response const & failure_response, char const * const error_message = nullptr) {
290 if (m_retries <= 0) {
291 (void)m_send_fw_state_callback(FW_STATE_FAILED, error_message);
292 m_fw_callback->Call_Callback<Logger>(false);
293 (void)m_finish_callback();
294 return;
295 }
296
297 // Decrease the amount of retries of downloads for the current chunk,
298 // reset as soon as the next chunk has been received and handled successfully
299 m_retries--;
300
301 switch (failure_response) {
302 case OTA_Failure_Response::RETRY_CHUNK:
303 Request_Next_Firmware_Packet();
304 break;
305 case OTA_Failure_Response::RETRY_UPDATE:
306 Request_First_Firmware_Packet();
307 break;
308 case OTA_Failure_Response::RETRY_NOTHING:
309 (void)m_send_fw_state_callback(FW_STATE_FAILED, error_message);
310 m_fw_callback->Call_Callback<Logger>(false);
311 (void)m_finish_callback();
312 break;
313 default:
314 // Nothing to do
315 break;
316 }
317 }
318
320 void Handle_Request_Timeout() {
321 uint64_t const & timeout = m_fw_callback->Get_Timeout();
322 char message[Helper::detectSize(CHUNK_REQUEST_TIMED_OUT, m_requested_chunks, timeout)] = {};
323 (void)snprintf(message, sizeof(message), CHUNK_REQUEST_TIMED_OUT, m_requested_chunks, timeout);
324 Logger::println(message);
325 Handle_Failure(OTA_Failure_Response::RETRY_CHUNK, message);
326 }
