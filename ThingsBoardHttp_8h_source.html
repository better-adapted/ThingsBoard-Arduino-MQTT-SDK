95 template <typename TSource>
96 bool Send_Json(char const * const topic, TSource const & source, size_t const & jsonSize) {
97 // Check if allocating needed memory failed when trying to create the JsonObject,
98 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
99 if (source.isNull()) {
100 Logger::println(UNABLE_TO_ALLOCATE_JSON);
101 return false;
102 }
103#if !THINGSBOARD_ENABLE_DYNAMIC
104 size_t const amount = source.size();
105 if (MaxFieldsAmount < amount) {
106 Logger::printfln(TOO_MANY_JSON_FIELDS, amount, MaxFieldsAmount);
107 return false;
108 }
109#endif // !THINGSBOARD_ENABLE_DYNAMIC
110 bool result = false;
111 if (getMaximumStackSize() < jsonSize) {
112 char * json = new char[jsonSize]();
113 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
114 Logger::println(UNABLE_TO_SERIALIZE_JSON);
115 }
116 else {
117 result = Send_Json_String(topic, json);
118 }
119 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
120 // and set the pointer to null so we do not have a dangling reference.
121 delete[] json;
122 json = nullptr;
123 }
124 else {
125 char json[jsonSize] = {};
126 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
127 Logger::println(UNABLE_TO_SERIALIZE_JSON);
128 return result;
129 }
130 result = Send_Json_String(topic, json);
131 }
132 return result;
133 }

139 bool Send_Json_String(char const * const topic, char const * const json) {
140 if (json == nullptr || m_token == nullptr) {
141 return false;
142 }
143
144 char path[Helper::detectSize(topic, m_token)] = {};
145 (void)snprintf(path, sizeof(path), topic, m_token);
146 return postMessage(path, json);
147 }

283 bool postMessage(char const * const path, char const * const json) {
284 bool success = m_client.post(path, HTTP_POST_PATH, json) == 0;
285 int const status = m_client.get_response_status_code();
286
287 if (!success || status < HTTP_RESPONSE_SUCCESS_RANGE_START || status > HTTP_RESPONSE_SUCCESS_RANGE_END) {
288 Logger::printfln(HTTP_FAILED, POST, status);
289 success = false;
290 }
291
292 clearConnection();
293 return success;
294 }

330 template<typename InputIterator>
331 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
332#if THINGSBOARD_ENABLE_DYNAMIC
333 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
334 // Data structure size depends on the amount of key value pairs passed.
335 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
336 size_t const size = Helper::distance(first, last);
337 TBJsonDocument jsonBuffer(JSON_OBJECT_SIZE(size));
338#else
339 StaticJsonDocument<JSON_OBJECT_SIZE(MaxFieldsAmount)> jsonBuffer;
340#endif // !THINGSBOARD_ENABLE_DYNAMIC
341
342 for (auto it = first; it != last; ++it) {
343 auto const & data = *it;
344 if (!data.SerializeKeyValue(jsonBuffer)) {
345 Logger::println(UNABLE_TO_SERIALIZE);
346 return false;
347 }
348 }
349
350 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
351 }
352
359 template<typename T>
360 bool sendKeyValue(char const * const key, T value, bool telemetry = true) {
361 Telemetry const t(key, value);
362 if (t.IsEmpty()) {
363 // Message is ignored and not sent at all.
364 return false;
365 }
366
367 StaticJsonDocument<JSON_OBJECT_SIZE(1)> jsonBuffer;
368 if (!t.SerializeKeyValue(jsonBuffer)) {
369 Logger::printfln(UNABLE_TO_SERIALIZE);
370 return false;
371 }
372
373 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
374 }
